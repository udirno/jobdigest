    ```javascript
    export function escapeHtml(text) {
      const div = document.createElement('div');
      div.textContent = text || '';
      return div.innerHTML;
    }

    export function extractKeyRequirements(description) {
      if (!description) return '';
      // Look for bullet points or sentences with requirement keywords
      const keywords = ['require', 'must have', 'experience with', 'proficien', 'knowledge of', 'skills in', 'familiar with', 'years of'];
      const sentences = description.split(/[.!?\n]+/).map(s => s.trim()).filter(s => s.length > 15);

      for (const sentence of sentences) {
        const lower = sentence.toLowerCase();
        if (keywords.some(kw => lower.includes(kw))) {
          return sentence.substring(0, 200);
        }
      }

      // Fallback: skip first sentence (usually generic intro), use second
      if (sentences.length > 1) {
        return sentences[1].substring(0, 150);
      }

      return (description || '').substring(0, 150);
    }

    export function formatRelativeDate(dateString) {
      if (!dateString) return '';
      const date = new Date(dateString);
      if (isNaN(date.getTime())) return dateString;
      const now = new Date();
      const diffMs = now - date;
      const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24));

      if (diffDays === 0) return 'Today';
      if (diffDays === 1) return 'Yesterday';
      if (diffDays < 7) return `${diffDays}d ago`;
      if (diffDays < 30) return `${Math.floor(diffDays / 7)}w ago`;
      return date.toLocaleDateString();
    }

    export function formatSalary(salary) {
      if (!salary) return null;
      // Handle string salary (legacy format)
      if (typeof salary === 'string') return salary;
      // Handle object salary { min, max }
      const parts = [];
      if (salary.min) parts.push(`$${Math.round(salary.min / 1000)}k`);
      if (salary.max) parts.push(`$${Math.round(salary.max / 1000)}k`);
      if (parts.length === 0) return null;
      if (parts.length === 1) return `${parts[0]}+`;
      return parts.join(' - ');
    }
    ```
