    **3. src/dashboard/job-modal.js** - Modal logic:

    ```javascript
    import { getFilteredAndSortedJobs } from './filters.js';
    import { escapeHtml, formatRelativeDate, formatSalary } from './job-card.js';
    ```

    Module state:
    - `let currentJobs = [];` (filtered/sorted array for navigation)
    - `let currentIndex = 0;`
    - `let modal, modalBody, prevBtn, nextBtn, counterEl, viewOriginalLink;`

    **initJobModal():**
    - Cache DOM references: modal = `document.getElementById('job-detail-modal')`, etc.
    - Attach click listeners to prevBtn and nextBtn
    - Attach keydown listener on the modal element for ArrowLeft/ArrowRight navigation
    - Attach `close` event on modal to clean up

    **openJobModal(jobId):**
    1. `currentJobs = getFilteredAndSortedJobs();`
    2. `currentIndex = currentJobs.findIndex(j => j.jobId === jobId);`
    3. If not found, return
    4. `renderModalContent(currentJobs[currentIndex]);`
    5. `updateNavigation();`
    6. `modal.showModal();`

    **renderModalContent(job):**
    Build modal body HTML using DOM manipulation or innerHTML (innerHTML is acceptable here since we escape all user content):

    - Job title (h2.modal-job-title)
    - Company (p.modal-job-company)
    - Meta row: location, date (formatRelativeDate), salary (formatSalary), source badge
    - **Score section** (only if job.score !== null && job.score >= 0):
      - Overall score (large number, use score color class)
      - AI reasoning (job.scoreReasoning) in `.score-reasoning` block
      - 5 dimension scores in `.dimension-grid`:
        - Skills Match: `job.scoreDetails?.skills_match`
        - Experience Level: `job.scoreDetails?.experience_level`
        - Tech Stack: `job.scoreDetails?.tech_stack_alignment`
        - Title Relevance: `job.scoreDetails?.title_relevance`
        - Industry Fit: `job.scoreDetails?.industry_fit`
      - Each dimension shows: name, numeric score, and a colored bar (width = score%)
      - Bar color: use same high/medium/low thresholds (>=80 green, >=60 amber, <60 red)
    - If unscored (score is null/undefined/-1): show message "This job has not been scored yet."
    - **Description section**: Full job description with newlines preserved (use `white-space: pre-wrap` and escape HTML, convert \n to proper whitespace)
    - Update `viewOriginalLink.href = job.url;`

    **updateNavigation():**
    - `prevBtn.disabled = currentIndex === 0;`
    - `nextBtn.disabled = currentIndex === currentJobs.length - 1;`
    - `counterEl.textContent = \`${currentIndex + 1} of ${currentJobs.length}\`;`

    **Navigation handlers:**
    - Previous: if currentIndex > 0, decrement, renderModalContent, updateNavigation
    - Next: if currentIndex < length - 1, increment, renderModalContent, updateNavigation
    - ArrowLeft in modal keydown: trigger previous (e.preventDefault())
    - ArrowRight in modal keydown: trigger next (e.preventDefault())

    **Dimension color helper:**
    ```javascript
    function getDimensionColor(score) {
      if (score >= 80) return '#81c784';  // green
      if (score >= 60) return '#ffd54f';  // amber
      return '#e57373';  // red
    }

    function getScoreClass(score) {
      if (score === null || score === undefined || score < 0) return 'unscored';
      if (score >= 80) return 'high';
      if (score >= 60) return 'medium';
      return 'low';
    }
    ```

    **New handler functions:**

    ```javascript
    async function handleClearJobs() {
      const statusEl = document.getElementById('data-mgmt-status');
      if (!confirm('Clear all jobs and scores? This cannot be undone.')) return;

      try {
        await storage.set('jobs', {});
        statusEl.className = 'status-indicator success';
        statusEl.textContent = 'All jobs cleared';
        setTimeout(() => { statusEl.className = 'status-indicator'; statusEl.textContent = ''; }, 3000);
      } catch (error) {
        statusEl.className = 'status-indicator error';
        statusEl.textContent = 'Failed to clear jobs';
      }
    }

    async function handleClearScores() {
      const statusEl = document.getElementById('data-mgmt-status');
      if (!confirm('Clear all scores? Jobs will be kept but scores removed.')) return;

      try {
        const jobs = await storage.getJobs();
        for (const jobId of Object.keys(jobs)) {
          jobs[jobId].score = null;
          jobs[jobId].scoreReasoning = null;
          jobs[jobId].scoredAt = null;
          jobs[jobId].scoreDetails = null;
        }
        await storage.set('jobs', jobs);
        statusEl.className = 'status-indicator success';
        statusEl.textContent = 'All scores cleared';
        setTimeout(() => { statusEl.className = 'status-indicator'; statusEl.textContent = ''; }, 3000);
      } catch (error) {
        statusEl.className = 'status-indicator error';
        statusEl.textContent = 'Failed to clear scores';
      }
    }

    async function handleResetSettings() {
      const statusEl = document.getElementById('data-mgmt-status');
      if (!confirm('Reset search preferences to defaults? API keys will be preserved.')) return;

      try {
        await storage.setSettings({
          fetchHour: 6,
          fetchMinute: 0,
          timezone: Intl.DateTimeFormat().resolvedOptions().timeZone,
          searchKeywords: [],
          location: '',
          salaryMin: null,
          datePosted: 'all',
          employmentType: 'FULLTIME',
          remoteOnly: false
        });
        statusEl.className = 'status-indicator success';
        statusEl.textContent = 'Settings reset to defaults';
        setTimeout(() => { statusEl.className = 'status-indicator'; statusEl.textContent = ''; }, 3000);

        // Refresh settings UI to show defaults
        const settingsContent = document.getElementById('settings-content');
        if (settingsContent) {
          await initSettings(settingsContent);
        }
      } catch (error) {
        statusEl.className = 'status-indicator error';
        statusEl.textContent = 'Failed to reset settings';
      }
    }
    ```
